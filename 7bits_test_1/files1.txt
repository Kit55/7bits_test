System.out.println();
while ((line = fin.readLine()) != null) {
text = text+line;
for (int i=0;i<line.length();i++) {

tabLvl-=1;
newLine(tabLvl);
}
System.out.print (line.charAt(i));
if ((line.charAt(i)==';') || (line.charAt(i)=='{')) {
if (line.charAt(i)=='{')
tabLvl+=1;
newLine(tabLvl);
}
}
}

}
